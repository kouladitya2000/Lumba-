$vaultName = "YOUR_KEYVAULT_NAME"

$secrets = Get-AzKeyVaultSecret -VaultName $vaultName

$secretDetails = foreach ($secret in $secrets) {
    
    $fullSecret = Get-AzKeyVaultSecret -VaultName $vaultName -Name $secret.Name

    [PSCustomObject]@{
        Name            = $fullSecret.Name
        VaultName       = $vaultName
        Version         = $fullSecret.Version
        Enabled         = $fullSecret.Enabled
        Created         = $fullSecret.Created
        Updated         = $fullSecret.Updated
        Expires         = $fullSecret.Expires
        NotBefore       = $fullSecret.NotBefore
        ContentType     = $fullSecret.ContentType
        Tags            = ($fullSecret.Tags | Out-String)
        SecretValue     = $fullSecret.SecretValueText
        Id              = $fullSecret.Id
    }
}

$secretDetails
