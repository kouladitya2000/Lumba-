param($Timer)

Import-Module AutomationModules/acst.az.autohelper
$currentUTCtime = (Get-Date).ToUniversalTime()
Write-Host "PowerShell timer trigger function ran! TIME: $currentUTCtime"

$password = ConvertTo-SecureString $env:application_secret -AsPlainText -Force
$TenantId = $env:tenant_id
$ApplicationId = $env:application_id
$Credential = New-Object -TypeName System.Management.Automation.PSCredential -ArgumentList $ApplicationId, $password
Disable-AzContextAutosave -Scope Process
try {
    Connect-AzAccount -ServicePrincipal -TenantId $TenantId -Subscription hsbc-multi-cstmg-01 -Credential $Credential
}
catch {
    Write-Host "Error connecting to Azure. $_.Exception"
    throw $_.Exception
}

#KQL Query
$KqlQuery = @"
resources
| where type =~ 'Microsoft.Compute/virtualMachines'
| where tags['Production'] =~ 'No'
| where tags['VM Type'] =~ 'Cattle'
| where tags['cattleAgeCompliance'] =~ 'Non-Compliant'
| where tags['VM Role'] != 'Appliance'
| where isempty(tags['DatabricksEnvironment'])
    and isempty(tags['databricks-environment'])
    and (isempty(tags['aks-managed-orchestrator']) or tags['aks-managed-orchestrator'] !startswith 'Kubernetes')
    and (isempty(tags['Vendor']) or tags['Vendor'] !contains 'Databricks')
| project id, name, resourceGroup, subscriptionId, location
"@

#Query Resource Graph
Write-Host "Querying Resource Graph for non-compliant cattle VMs..."

try {
    $vmList = Search-AzGraph -Query $KqlQuery -First 1000
}
catch {
    $functionName = $MyInvocation.MyCommand.Name
    $errorMessage = "Error querying Resource Graph in function '$functionName': $($_.Exception.Message)"
    Write-Host "Error querying Resource Graph. $($_.Exception.Message)"
    Email-Alert -FunctionName $functionName -ErrorMessage $errorMessage
    throw $_.Exception
}

Write-Host "Fetched $(@($vmList).Count) non-compliant VM(s)"

#VM List data print
Foreach ($vm in $vmList) {
    Write-Host "VM Name: $($vm.name), Resource Group: $($vm.resourceGroup), Subscription ID: $($vm.subscriptionId), Location: $($vm.location)"
}

#Shutdown Loop
if ($vmList.Count -gt 0) {

    foreach ($vm in $vmList) {
        $logCtx = "VM=[$($vm.name)] RG=[$($vm.resourceGroup)] Sub=[$($vm.subscriptionId)]"
        Write-Host "Processing: $logCtx"

        # Switch context to VM's subscription to fetch power state
        Set-AzContext -SubscriptionId $vm.subscriptionId | Out-Null

        # Check power state - only proceed if VM is running
        $powerState = (Get-AzVM -ResourceGroupName $vm.resourceGroup -Name $vm.name -Status).Statuses |
                      Where-Object { $_.Code -like 'PowerState/*' } |
                      Select-Object -ExpandProperty Code

        if ($powerState -ne 'PowerState/running') {
            Write-Host "⏭ VM not running (current state: $powerState), skipping: $logCtx"
            continue
        }

        try {
            Stop-AzVM `
                -Name              $vm.name `
                -ResourceGroupName $vm.resourceGroup `
                -Force `
                -NoWait

            Write-Host "🟩 Shutdown initiated: $logCtx"
        }
        catch {
            $functionName = $MyInvocation.MyCommand.Name
            $errorMessage = "Error shutting down VM in function '$functionName': $logCtx – $($_.Exception.Message)"
            Write-Host "Error shutting down VM. $($_.Exception.Message)"
            Email-Alert -FunctionName $functionName -ErrorMessage $errorMessage
            # Don't throw – continue to next VM
        }
    }

    Write-Host "🟩 Done. All non-compliant cattle VMs processed." -ForegroundColor Green

} else {
    $functionName = $MyInvocation.MyCommand.Name
    $errorMessage = "Non-compliant cattle VM list is empty and returned 0 results. Nothing shut down. Function '$functionName'"
    Write-Host "⚠ No non-compliant VMs found. Nothing to shut down." -ForegroundColor Yellow
    Email-Alert -FunctionName $functionName -ErrorMessage $errorMessage
}
