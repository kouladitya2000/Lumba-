param(
    [Parameter(Mandatory=$true)]
    [string]$SubscriptionId,

    [Parameter(Mandatory=$true)]
    [string]$ResourceGroup,

    [Parameter(Mandatory=$true)]
    [string]$AseName
)

# ---------- LOGIN ----------
Connect-AzAccount -ErrorAction Stop
Set-AzContext -SubscriptionId $SubscriptionId -ErrorAction Stop

$apiVersion = "2022-09-01"

$uri = "https://management.azure.com/subscriptions/$SubscriptionId/resourceGroups/$ResourceGroup/providers/Microsoft.Web/hostingEnvironments/$AseName?api-version=$apiVersion"

Write-Host "Fetching ASE..." -ForegroundColor Cyan

$response = Invoke-AzRestMethod -Method GET -Uri $uri -ErrorAction Stop
$ase = $response.Content | ConvertFrom-Json

# ---------- VALIDATIONS ----------

if ($ase.kind -notmatch "ASEV3") {
    throw "This resource is NOT ASE v3. Detected kind: $($ase.kind)"
}

if (-not $ase.properties.maximumNumberOfZones -or $ase.properties.maximumNumberOfZones -lt 2) {
    throw "This region does not support Availability Zones for ASE."
}

Write-Host "ASE Version: $($ase.kind)"
Write-Host "Region: $($ase.location)"
Write-Host "Max Zones Supported: $($ase.properties.maximumNumberOfZones)"
Write-Host "Current Zone Redundancy: $($ase.properties.zoneRedundancyEnabled)"

# ---------- IDEMPOTENT CHECK ----------

if ($ase.properties.zoneRedundancyEnabled -eq $true) {
    Write-Host "Zone redundancy already enabled. Nothing to do." -ForegroundColor Green
    return
}

Write-Host "Enabling Zone Redundancy..." -ForegroundColor Yellow

# ---------- UPDATE ----------
$ase.properties.zoneRedundancyEnabled = $true

Invoke-AzRestMethod `
    -Method PUT `
    -Uri $uri `
    -Payload ($ase | ConvertTo-Json -Depth 30) `
    -ErrorAction Stop

Write-Host "Zone redundancy update submitted successfully." -ForegroundColor Green
Write-Host "Operation can take 12–24 hours to complete."
