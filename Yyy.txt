    $triggeredFunctions = @()

    foreach ($missingFile in $missingFiles) {

        if ($fileFunctionMap.ContainsKey($missingFile)) {

            $functionName = $fileFunctionMap[$missingFile]

            if ($triggeredFunctions -notcontains $functionName) {

                $triggeredFunctions += $functionName

                $url = "https://$functionAppName.azurewebsites.net/admin/functions/$functionName"

                Invoke-WebRequest -Uri $url -Headers $header -Method Post -Body $bodyJson

                Write-Host "Triggered function: $functionName"
                Write-Host "Starting Sleep before next trigger"

                Start-Sleep -Seconds 240
            }
        }
    }
}
else {
    Write-Output "All files present"
}
