$subscriptionId = "<subscription-id>"
$resourceGroup  = "<resource-group>"
$aseName        = "<ase-name>"

Connect-AzAccount
Set-AzContext -SubscriptionId $subscriptionId

$apiVersion = "2022-09-01"

$path = "/subscriptions/$subscriptionId/resourceGroups/$resourceGroup/providers/Microsoft.Web/hostingEnvironments/$aseName"

# Get ASE
$response = Invoke-AzRestMethod `
    -Method GET `
    -Path $path `
    -ApiVersion $apiVersion

$ase = $response.Content | ConvertFrom-Json

Write-Host "Kind:" $ase.kind
Write-Host "Current ZoneRedundancyEnabled:" $ase.properties.zoneRedundancyEnabled
Write-Host "Max Zones Supported:" $ase.properties.maximumNumberOfZones

# Validation
if ($ase.kind -ne "ASEV3") {
    throw "This is not ASE v3"
}

if ($ase.properties.maximumNumberOfZones -lt 2) {
    throw "Region does not support availability zones"
}

if ($ase.properties.zoneRedundancyEnabled -eq $true) {
    Write-Host "Already enabled. Nothing to do." -ForegroundColor Green
    return
}

# PATCH only the required property
$body = @{
    properties = @{
        zoneRedundancyEnabled = $true
    }
} | ConvertTo-Json -Depth 5

Invoke-AzRestMethod `
    -Method PATCH `
    -Path $path `
    -ApiVersion $apiVersion `
    -Payload $body

Write-Host "Zone redundancy update submitted successfully." -ForegroundColor Green
Write-Host "This can take 12–24 hours."
