$storageAccountName = "yourstorageaccount"
$containerName = "yourcontainer"
$filePath = "C:\temp\file.txt"

# Get storage context
$ctx = (Get-AzStorageAccount -Name $storageAccountName -ResourceGroupName "your-rg").Context

# Upload file
Set-AzStorageBlobContent `
    -File $filePath `
    -Container $containerName `
    -Blob "file.txt" `
    -Context $ctx
