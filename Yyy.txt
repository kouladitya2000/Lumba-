param($Timer)

Import-Module AutomationModules/acst.az.autohelper

# Get the current universal time in the default string format
$currentUTCtime = (Get-Date).ToUniversalTime()

# Write an information log with the current time.
Write-Host "PowerShell timer trigger function ran! TIME: $currentUTCtime"

try {
    Connect-AzAccount -Identity -AccountId $env:MSI_CLIENT_ID
}
catch {
    Write-Host "Error connecting to Azure. $_.Exception"
    throw $_.Exception
}

$todayDate = Get-Date -Format "dddd, MMMM dd, yyyy"
Write-Host "Hello! Today's date is $todayDate"
