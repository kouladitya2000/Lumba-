resource "azurerm_storage_account" "storage" {
  name                              = replace(local.deploy_name, "-", "")
  location                          = azurerm_resource_group.automation.location
  resource_group_name               = azurerm_resource_group.automation.name
  account_tier                      = "Standard"
  account_replication_type          = var.globals.storage_replication
  min_tls_version                   = "TLS1_2"
  https_traffic_only_enabled        = true
  allow_nested_items_to_be_public   = false
  cross_tenant_replication_enabled  = false
  default_to_oauth_authentication   = true
  public_network_access_enabled     = false
  infrastructure_encryption_enabled = true
  shared_access_key_enabled         = true

  blob_properties {
    versioning_enabled            = true
    change_feed_enabled           = true
    change_feed_retention_in_days = 7
    delete_retention_policy {
      days = 7
    }
    container_delete_retention_policy {
      days = 7
    }
  }

  network_rules {
    default_action = "Deny"
    bypass         = ["Logging", "Metrics", "AzureServices"]
    subnet_ids     = [azurerm_subnet.functions.id]
  }

  identity {
    type         = "SystemAssigned, UserAssigned"
    identity_ids = [azurerm_user_assigned_identity.automation.id]
  }

  tags = var.globals.tags

  lifecycle {
    ignore_changes = [tags]
  }
}

resource "azurerm_private_endpoint" "storage_pe" {
  for_each            = var.globals.storage_services
  name                = "${replace(local.deploy_name, "-", "")}-${each.value.name}-pep"
  location            = azurerm_resource_group.automation_endpoints.location
  resource_group_name = azurerm_resource_group.automation_endpoints.name
  subnet_id           = data.azurerm_subnet.pe_subnet.id

  private_service_connection {
    name                           = "${replace(local.deploy_name, "-", "")}-${each.value.name}-psc"
    private_connection_resource_id = azurerm_storage_account.storage.id
    is_manual_connection           = false
    subresource_names              = [each.value.name]
  }

  private_dns_zone_group {
    name                 = "default"
    private_dns_zone_ids = [each.value.private_dns_zone_id]
  }

  lifecycle {
    ignore_changes = [tags]
  }
}
