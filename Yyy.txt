function UpdateConfluencePage($baseUrl, $pageId, $username, $apiToken, $data){

    # Encode auth
    $encoded = [Convert]::ToBase64String(
        [Text.Encoding]::ASCII.GetBytes("$($username):$($apiToken)")
    )

    # --- SSL BYPASS HANDLER ---
    Add-Type @"
using System.Net.Http;
using System.Net.Security;
using System.Security.Cryptography.X509Certificates;

public class UnsafeHttpHandler : HttpClientHandler
{
    public UnsafeHttpHandler()
    {
        ServerCertificateCustomValidationCallback =
            HttpClientHandler.DangerousAcceptAnyServerCertificateValidator;
    }
}
"@

    $handler = [UnsafeHttpHandler]::new()
    $client = [System.Net.Http.HttpClient]::new($handler)

    $client.DefaultRequestHeaders.Clear()
    $client.DefaultRequestHeaders.Add("Authorization", "Basic $encoded")
    $client.DefaultRequestHeaders.Add("Accept", "application/json")

    # ==========================
    # 1️⃣ GET EXISTING PAGE
    # ==========================
    $getUri = "$baseUrl/confluence/rest/api/content/$pageId"

    Write-Host "Calling GET: $getUri"

    $getResponse = $client.GetAsync($getUri).GetAwaiter().GetResult()

    if (-not $getResponse.IsSuccessStatusCode) {
        throw "GET failed: $($getResponse.StatusCode)"
    }

    $getContent = $getResponse.Content.ReadAsStringAsync().GetAwaiter().GetResult()
    $response = $getContent | ConvertFrom-Json

    $version = $response.version.number + 1

    # ==========================
    # 2️⃣ PREPARE UPDATE BODY
    # ==========================
    $payload = @{
        id = $pageId
        type = "page"
        title = $response.title
        space = @{
            key = $response.space.key
        }
        body = @{
            storage = @{
                value = GetHtmlTable($data)
                representation = "storage"
            }
        }
        version = @{
            number = $version
        }
    } | ConvertTo-Json -Depth 10

    $putUri = "$baseUrl/confluence/rest/api/content/$pageId"

    Write-Host "Calling PUT: $putUri"

    $contentObj = New-Object System.Net.Http.StringContent($payload, [Text.Encoding]::UTF8, "application/json")

    $putResponse = $client.PutAsync($putUri, $contentObj).GetAwaiter().GetResult()

    if (-not $putResponse.IsSuccessStatusCode) {
        throw "PUT failed: $($putResponse.StatusCode)"
    }

    Write-Host "Page updated successfully."
}
