function Send-GuardiumEmail {
    param (
        [string]$subject,
        [string]$body
    )

    $emails = $EmailAddressGuardiumChain -split ","

    foreach ($email in $emails) {
        try {
            $emailReqObject = @{
                to      = $email.Trim()
                subject = $subject
                body    = $body
                isHtml  = $true
            }

            $emailReqBody = $emailReqObject | ConvertTo-Json -Depth 5

            Invoke-WebRequest `
                -Uri $emailNotificationUri `
                -Method POST `
                -Body $emailReqBody `
                -ContentType "application/json" `
                -ErrorAction Stop
        }
        catch {
            Write-Warning "Failed to send email to $email"
        }
    }
}
