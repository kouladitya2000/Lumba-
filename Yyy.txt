param(
    [string]$Environment,
    [string]$ResourceGroupName,
    [string]$ASEName
)

Describe "App Service Environment Validation - $Environment" {

    BeforeAll {
        Write-Host "Checking ASE in environment: $Environment"
        Write-Host "Resource Group: $ResourceGroupName"
        Write-Host "ASE Name: $ASEName"

        try {
            $ase = az appservice ase show `
                --name $ASEName `
                --resource-group $ResourceGroupName `
                --query "name" `
                --output tsv 2>$null
        }
        catch {
            $ase = $null
        }
    }

    Context "ASE Existence Check" {

        It "ASE should exist in resource group $ResourceGroupName" {

            $ase | Should -Not -BeNullOrEmpty
            $ase | Should -Be $ASEName

        }

    }

}
