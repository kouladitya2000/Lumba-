param($Timer)

$Remote_Tenant_Name = "sitecore"
$Remote_Tenant_ID = "0d210810-df46-4261-9c00-1189902ff0f8"

$Strategic_Tenant_ID = "e0fd434d-ba64-497b-90d2-859c472e1a92"
$Strategic_KV_Tenant_Mgt = "labubumulticstmg-es-pd-kv"
$Strategic_KV_Tenant_Mgt_SubID = "9af5e111-b019-411a-9729-02309840da84"

try {
    Write-Output "Connect to Azure using the Automation Account system assigned managed identity"
    Set-Item Env:\SuppressAzurePowerShellBreakingChangeWarnings "true" -InformationAction SilentlyContinue
    Clear-AzContext -Scope CurrentUser -Force -ErrorAction SilentlyContinue -InformationAction SilentlyContinue
    Connect-AzAccount -Identity -AccountId $env:MSI_CLIENT_ID | Out-Null

    Write-Output "Select Strategic SubID $($Strategic_KV_Tenant_Mgt_SubID) where keyvault with Management creds are stored"
    Set-AzContext -SubscriptionId $Strategic_KV_Tenant_Mgt_SubID -TenantId $Strategic_Tenant_ID | Out-Null

    Write-Output "Get SPN secret from kv $($Strategic_KV_Tenant_Mgt)"
    if ($Remote_Tenant_Name -eq "Strategic") {
        $secret = (Get-AzKeyVaultSecret -VaultName $Strategic_KV_Tenant_Mgt -Name "ARMCLIENTSECRET").SecretValue
        $Remote_SPN_ID = (Get-AzKeyVaultSecret -VaultName $Strategic_KV_Tenant_Mgt -Name "ARMCLIENTID").SecretValueText
    }
    else {
        $secret = (Get-AzKeyVaultSecret -VaultName $Strategic_KV_Tenant_Mgt -Name "$($Remote_Tenant_Name)-ARMCLIENT-Secret").SecretValue
        $Remote_SPN_ID = (Get-AzKeyVaultSecret -VaultName $Strategic_KV_Tenant_Mgt -Name "$($Remote_Tenant_Name)-ARMCLIENTID").SecretValueText
    }

    Write-Output "Create PS Credential object to login to target tenant using SPN"
    $loginCred = (New-Object System.Management.Automation.PSCredential -ArgumentList $Remote_SPN_ID, $secret)

    Write-Output "Connecting to target tenant: $($Remote_Tenant_Name)"
    Clear-AzContext -Scope CurrentUser -Force -ErrorAction SilentlyContinue -InformationAction SilentlyContinue
    Connect-AzAccount -ServicePrincipal -TenantId $Remote_Tenant_ID -Credential $loginCred -InformationAction SilentlyContinue

    function get-authHeader {
        try {
            $azContext = Get-AzContext
            $azProfile = [Microsoft.Azure.Commands.Common.Authentication.Abstractions.AzureRmProfileProvider]::Instance.Profile
            $profileClient = New-Object -TypeName Microsoft.Azure.Commands.ResourceManager.Common.RMProfileClient -ArgumentList ($azProfile)
            $token = $profileClient.AcquireAccessToken($azContext.Subscription.TenantId)
            $authHeader = @{
                'Content-Type'  = 'application/json'
                'Authorization' = 'Bearer ' + $token.AccessToken
            }
            return $authHeader
        }
        catch {
            Write-Error "🔴 - Error getting auth header: Error: $($_.Exception)"
            throw $_.Exception
        }
    }

    function runQuery {
        Param(
            [string][Parameter(Mandatory = $true)] $query
        )
        try {
            $authHeader = get-authHeader
            $top = 1000
            $skipToken = $null
            $response = $null
            [System.Collections.ArrayList]$Global:results = @()
            do {
                $request = @{
                    Uri     = 'https://management.azure.com/providers/Microsoft.ResourceGraph/resources?api-version=2021-03-01'
                    Method  = 'POST'
                    Headers = $authHeader
                    Body    = @{
                        query   = $query
                        options = if (!$skipToken) {
                            @{ '$top' = $top }
                        }
                        else {
                            @{ '$top' = $top; '$skipToken' = $skipToken }
                        }
                    } | ConvertTo-Json -Depth 5
                }
                $response = Invoke-RestMethod @request -UseBasicParsing
                $skipToken = $response.'$SkipToken'
                $results.AddRange($response.data)
            } while ($skipToken)
            return $results
        }
        catch {
            Write-Error "🔴 - Error running runQuery: $query, Error: $($_.Exception)"
            throw $_.Exception
        }
    }

    function Get-appGws {
        $result = @()
        try {
            $Query = @'
resources
| where type == "microsoft.network/applicationgateways"
| project name, id
'@
            $list = runQuery -Query $Query
            $result += $list
            return $result
        }
        catch {
            Write-Error "🔴 - Error getting Application Gateway List, Error: $($_.Exception)"
            throw $_.Exception
        }
    }

    $authHeader = get-authHeader
    $appGws = Get-appGws
    $tagName = "PublicWAF"

    foreach ($appGw in $appGws) {
        $appGwName = $appGw.name
        $public = "No"

        $appGwDetailResponse = Invoke-RestMethod -Method Get -Uri "https://management.azure.com$($appGw.id)?api-version=2024-01-01" -Headers $authHeader

        foreach ($frontendIP in $appGwDetailResponse.properties.frontendIPConfigurations) {
            if ($frontendIP.properties.publicIPAddress) {
                foreach ($listener in $appGwDetailResponse.properties.httpListeners) {
                    if ($listener.properties.frontendIPConfiguration.id -eq $frontendIP.id) {
                        $public = "True"
                    }
                }
            }
        }

        if ("True" -eq $public) {
            Write-Host "Application Gateway $appGwName is publicly accessible." -ForegroundColor Red
        }
        else {
            Write-Host "Application Gateway $appGwName is not publicly accessible." -ForegroundColor Green
        }

        $currentTags = @{}
        if ($appGwDetailResponse.tags) {
            foreach ($tag in $appGwDetailResponse.tags.PSObject.Properties) {
                $currentTags[$tag.Name] = $tag.Value
            }
        }

        $desiredTagValue = if ("True" -eq $public) { "Yes" } else { "No" }
        $currentTagValue = $currentTags[$tagName]

        if ($null -eq $currentTagValue) {
            Write-Host "Tag '$tagName' is missing for '$appGwName'. Adding tag with PUT."
            $currentTags[$tagName] = $desiredTagValue
            $body = @{
                properties = @{
                    tags = $currentTags
                }
            } | ConvertTo-Json -Depth 4
            Invoke-RestMethod -Method PUT -Uri "https://management.azure.com$($appGw.id)/providers/Microsoft.Resources/tags/default?api-version=2021-04-01" -Headers $authHeader -Body $body -ContentType "application/json"
            Write-Host "Tag '$tagName' added to '$appGwName' with value '$desiredTagValue'."
        }
        elseif ($currentTagValue -ne $desiredTagValue) {
            Write-Host "Tag '$tagName' value for '$appGwName' is '$currentTagValue'. Updating to '$desiredTagValue' with PATCH."
            $patchTags = @{$tagName = $desiredTagValue}
            $body = @{
                operation  = "Merge"
                properties = @{
                    tags = $patchTags
                }
            } | ConvertTo-Json -Depth 4
            Invoke-RestMethod -Method PATCH -Uri "https://management.azure.com$($appGw.id)/providers/Microsoft.Resources/tags/default?api-version=2021-04-01" -Headers $authHeader -Body $body -ContentType "application/json"
            Write-Host "Tag '$tagName' updated for '$appGwName' to '$desiredTagValue'."
        }
        else {
            Write-Host "Tag '$tagName' for '$appGwName' is already set to '$desiredTagValue'. No update needed."
        }
    }
}
catch {
    Write-Error "🔴 - Error connecting: $($_.Exception)"
}
finally {
    Clear-AzContext -Scope CurrentUser -Force -ErrorAction SilentlyContinue
}
