3️⃣ ase.tf

# Resource Group
resource "azurerm_resource_group" "rg" {
  name     = var.resource_group_name
  location = var.location
  tags     = var.tags
}

# Get existing VNet
data "azurerm_virtual_network" "vnet" {
  name                = var.vnet_name
  resource_group_name = var.vnet_rg
}

# Get ASE subnet
data "azurerm_subnet" "ase_subnet" {
  name                 = var.subnet_name
  virtual_network_name = var.vnet_name
  resource_group_name  = var.vnet_rg
}

# App Service Environment v3
resource "azurerm_app_service_environment_v3" "ase" {
  name                         = var.ase_name
  resource_group_name          = azurerm_resource_group.rg.name
  location                     = var.location
  subnet_id                    = data.azurerm_subnet.ase_subnet.id
  internal_load_balancing_mode = var.internal_load_balancing_mode
  zone_redundant               = var.zone_redundant

  tags = var.tags
}

Clean. Nothing hardcoded.

🚀 HOW TO RUN LOCALLY

Go to:

cd platform/ase

Then run:

terraform init -backend-config=backend-nonprod.hcl
terraform plan -var-file=nonprod.tfvars
terraform apply -var-file=nonprod.tfvars

That’s it.

⸻

🚀 Minimal Azure DevOps Pipeline

Create: pipelines/ase-platform.yml

trigger: none

parameters:
- name: environment
  type: string
  default: nonprod

stages:
- stage: Deploy_ASE
  jobs:
  - job: Terraform
    pool:
      name: $(AGENT_POOL)

    steps:
    - checkout: self

    - script: |
        cd platform/ase

        terraform init -backend-config=backend-${{ parameters.environment }}.hcl
        terraform plan -var-file=${{ parameters.environment }}.tfvars
        terraform apply -auto-approve -var-file=${{ parameters.environment }}.tfvars

Pipeline contains:
	•	No infra values
	•	No backend values
	•	Only environment selector

Exactly what you wanted.
