# Connect if required
Connect-AzAccount

# Managed Identity Object (Principal) ID from Function App -> Identity
$PrincipalId = "b412d32b-cc89-44ec-8ec8-bdb1e6d125ed"

# Application Insights details
$AppInsightsName = "<APP_INSIGHTS_NAME>"
$AppInsightsRG   = "<APP_INSIGHTS_RESOURCE_GROUP>"

# Get App Insights Resource ID
$AppInsightsId = (Get-AzResource `
    -ResourceGroupName $AppInsightsRG `
    -ResourceType "Microsoft.Insights/components" `
    -Name $AppInsightsName).ResourceId

# Assign Monitoring Metrics Publisher role
New-AzRoleAssignment `
    -ObjectId $PrincipalId `
    -RoleDefinitionName "Monitoring Metrics Publisher" `
    -Scope $AppInsightsId

Write-Host "Monitoring Metrics Publisher role assigned successfully."
