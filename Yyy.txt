$subscriptionId = "<subscription-id>"
$resourceGroup  = "<resource-group>"
$aseName        = "<ase-name>"

az account set --subscription $subscriptionId

$apiVersion = "2022-09-01"

$resourceId = "/subscriptions/$subscriptionId/resourceGroups/$resourceGroup/providers/Microsoft.Web/hostingEnvironments/$aseName"

# Check current state
az rest --method GET `
  --uri "https://management.azure.com$resourceId?api-version=$apiVersion"

# Enable zone redundancy
az rest --method PATCH `
  --uri "https://management.azure.com$resourceId?api-version=$apiVersion" `
  --body "{ `"properties`": { `"zoneRedundancyEnabled`": true } }"

Write-Host "Zone redundancy update submitted."
Write-Host "Operation can take 12–24 hours."
