# ===== VARIABLES =====
$subscriptionId = "<subscription-id>"
$resourceGroup = "<resource-group>"
$aseName = "stmg-nonprod-euw-mon-ase-01"
$apiVersion = "2023-12-01"

# ===== LOGIN & CONTEXT =====
Connect-AzAccount
Set-AzContext -SubscriptionId $subscriptionId

# ===== GET CURRENT ASE =====
$getResponse = Invoke-AzRestMethod `
    -Method GET `
    -Path "/subscriptions/$subscriptionId/resourceGroups/$resourceGroup/providers/Microsoft.Web/hostingEnvironments/$aseName?api-version=$apiVersion"

$ase = ($getResponse.Content | ConvertFrom-Json)

# ===== SAFETY CHECK =====
if ($ase.properties.maximumNumberOfZones -lt 2) {
    Write-Host "Region/Subnet does not support availability zones. Cannot enable zone redundancy."
    return
}

# ===== ENABLE ZONE REDUNDANCY =====
$ase.properties.zoneRedundant = $true

# ===== UPDATE ASE =====
Invoke-AzRestMethod `
    -Method PUT `
    -Path "/subscriptions/$subscriptionId/resourceGroups/$resourceGroup/providers/Microsoft.Web/hostingEnvironments/$aseName?api-version=$apiVersion" `
    -Payload ($ase | ConvertTo-Json -Depth 25)

Write-Host "Zone redundancy update triggered successfully."
