# Storage info
$subscriptionId = "<subscription-id>"
$storageAccountName = "<storage-account>"
$containerName = "waf"

# Login context
$ctx = (Get-AzStorageAccount -Name $storageAccountName).Context

# Get all tenant summary html files
$summaryFiles = Get-AzStorageBlob -Container $containerName -Context $ctx |
Where-Object { $_.Name -like "AppGwWaf_Compare_SummaryBySubscription_*.html" }

$tenantNav = @()

foreach ($file in $summaryFiles) {

    $tenant = $file.Name `
        -replace "AppGwWaf_Compare_SummaryBySubscription_" `
        -replace ".html"

    $tenantNav += [PSCustomObject]@{
        label       = $tenant
        tenantid    = $tenant
        summary     = "AppGwWaf_Compare_SummaryBySubscription_$tenant.html"
        summaryCSV  = "AppGwWaf_Compare_SummaryBySubscription_$tenant.csv"
        detailsHtml = "AppGwWaf_Compare_Details_$tenant.html"
        detailsCSV  = "AppGwWaf_Compare_Details_$tenant.csv"
    }
}

# Convert to JSON
$tenantJson = $tenantNav | ConvertTo-Json -Depth 5 -Compress

# Build index html
$indexHtml = @"
<html>
<head>
<meta charset='utf-8'>
<title>AppGw WAF Compare - All Tenants</title>
<style>
body{font-family:Arial,Helvetica,sans-serif;font-size:12px}
.bar{display:flex;gap:8px;align-items:center;flex-wrap:wrap;margin-bottom:10px}
button{padding:6px 10px}
.mono{font-family:Consolas,Monaco,monospace}
iframe{width:100%;height:82vh;border:1px solid #ccc}
</style>
</head>

<body>

<h2>All Tenants - Subscription Summary</h2>

<div class="bar">
<button id='prevBtn' onclick='prevTenant()'>Previous</button>
<button id='nextBtn' onclick='nextTenant()'>Next</button>
<span id='pos'></span>
<span class='mono' id='tenantLabel'></span>

<a id='openSummary' href='#' target='_blank' style='margin-left:10px'>Open summary</a>
<a id='openDetailsHtml' href='#' target='_blank'>Open details (HTML)</a>
<a id='openSummaryCsv' href='#' target='_blank'>Summary CSV</a>
<a id='openDetailsCsv' href='#' target='_blank'>Details CSV</a>
</div>

<iframe id='frame' src='about:blank'></iframe>

<script>

const tenantsRaw = $tenantJson
const tenants = Array.isArray(tenantsRaw) ? tenantsRaw : [tenantsRaw]

let idx = 0

function render(){

if(!tenants || tenants.length === 0){
document.getElementById('tenantLabel').innerText='No tenants found'
return
}

if(idx < 0) idx = 0
if(idx > tenants.length-1) idx = tenants.length-1

const t = tenants[idx]

document.getElementById('tenantLabel').innerText = t.label
document.getElementById('pos').innerText = (idx+1) + " of " + tenants.length

document.getElementById('frame').src = t.summary

document.getElementById('openSummary').href = t.summary
document.getElementById('openDetailsHtml').href = t.detailsHtml
document.getElementById('openSummaryCsv').href = t.summaryCSV
document.getElementById('openDetailsCsv').href = t.detailsCSV

document.getElementById('prevBtn').disabled = idx === 0
document.getElementById('nextBtn').disabled = idx === tenants.length-1
}

function prevTenant(){
idx--
render()
}

function nextTenant(){
idx++
render()
}

render()

</script>

</body>
</html>
"@

# Save locally
$tmp = Join-Path $env:TEMP "index.html"
[System.IO.File]::WriteAllText($tmp,$indexHtml)

# Upload to storage
Set-AzStorageBlobContent `
    -Container $containerName `
    -File $tmp `
    -Blob "index.html" `
    -Context $ctx `
    -Force

Write-Host "Index rebuilt successfully"
