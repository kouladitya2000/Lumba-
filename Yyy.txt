param($Timer)

Write-Host "[$(Get-Date -Format 'u')] Starting cleanup of `$web container..."

try {
    $storageAccountName = $env:STORAGE_ACCOUNT_NAME

    if (-not $storageAccountName) {
        throw "Missing STORAGE_ACCOUNT_NAME in app settings."
    }

    $ctx = New-AzStorageContext `
        -StorageAccountName $storageAccountName `
        -UseConnectedAccount

    $containerName = '$web'

    # Safety check
    if ($containerName -ne '$web') {
        throw "Safety check failed: only `$web container is allowed."
    }

    $deletedCount = 0
    $skippedCount = 0

    Get-AzStorageBlob -Container $containerName -Context $ctx | ForEach-Object {

        if ($_.Name -eq "error.html") {
            Write-Host "Skipping protected blob: $($_.Name)"
            $skippedCount++
            return
        }

        Write-Host "Deleting blob: $($_.Name)"

        Remove-AzStorageBlob `
            -Container $containerName `
            -Blob $_.Name `
            -Context $ctx `
            -Force

        $deletedCount++
    }

    Write-Host "✅ Cleanup completed. Deleted: $deletedCount | Skipped: $skippedCount"
}
catch {
    Write-Error "❌ Cleanup failed: $_"
    throw
}
